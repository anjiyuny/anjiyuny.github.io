### 📂 Array에 자주쓰이는 다양한 함수

<br>

#### sort()

배열을 정리하고 싶을때 쓸 수 있는 함수이다.








```js
   var arr = [7, 3, 5, 2, 40];
    // arr.sort(function(a, b){
    //     return a -b
    //     // 숫자 오름차순 정렬
    //     //  1. a,b는 array안의 자료
    //     //  2. return 오른쪽이 양수면 a를 오른쪽으로 
    //     //  3. return 오른쪽이 음수면 b를 오른쪽으로 
    // }); 
    // // sort() 문자순으로 정렬함
    var newArr = arr.filter(function(a){ 
    // filter() 결과는 변수에 저장해서 써야함
        return a < 4
    })
    var newArr2 = arr.map(function(a){ 
    // filter() 결과는 변수에 저장해서 써야함
        return a * 4
    })
    
    console.log(newArr2);
    // sort() 원본을 수정함
    // filter() 원본 변형 안함


    // 이름순 정렬
    var arr = ['다', '가', '나'];
    arr.sort(function(a, b){
        return a < b ? 1:-1
    });
    //arr.sort();   // 가나다 정렬
    console.log(arr)
```
