### 📂 Lotto Game

```js
    var lottoArr = [];

    for(let i = 0; i < 6; i++){

        var num = Math.random() * 45 + 1;
        var parse = parseInt(num);

        lottoArr.forEach((data)=>{
            if(parse == data){
                parse = parseInt(num);
            }
        });
        lottoArr.push(parse); // 배열에 추가
    }
    // lottoArr.sort((a,b)=> a-b);
    console.log(lottoArr)
    
    document.write('이번주 로또 번호는 : ' + lottoArr)
```
